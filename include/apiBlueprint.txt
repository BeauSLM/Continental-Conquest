Endpoint      : 1
Description   : Get Player Information
URL           : http://localhost:80/api/player
Method        : GET
Input         : playerName
Output        : [{
					"AccountID": int,
					"Username": string,
					"Playetime": string,
					"Sub_status": string,
					"Guild": string,
					"bod": date
				}]
